class ManageSearchClass {

    // Associations
    private SearchScreen searchScreen;
    private PendingOrderScreen pendingOrderScreen;
    private DBManager dbManager;

    public void init() {

        this.dbManager = new DBManager();

        this.searchScreen = new SearchScreen(this);

        searchScreen.display();
    }

    public void searchOrders() {

        Object results = dbManager.queryOrders();

        pendingOrderScreen =
                new PendingOrderScreen(this);

        pendingOrderScreen.showResults(results);
    }
}

        System.out.println(

class ManageOrderClass {

    // Associations
    private OrderScreen orderScreen;
    private DBManager dbManager;

    public ManageOrderClass(
            OrderScreen orderScreen,
            DBManager dbManager) {

        this.orderScreen = orderScreen;
        this.dbManager = dbManager;
    }

    public void manageOrder() {

        boolean available =
                checkAvailability();

        if (available) {

            reduceQuantity();

            validateOrder();

            OrderUpdateScreen screen =
                    new OrderUpdateScreen();

            screen.confirmOrder();

        } else {

            NoAvailableProductScreen screen =
                    new NoAvailableProductScreen();

            screen.display();
        }
    }

    public boolean checkAvailability() {

        return dbManager.checkQuantity();
    }

    public void reduceQuantity() {

        dbManager.reduceQuantity();
    }

    public void validateOrder() {

        dbManager.validateOrderDetails();
    }
}

class Order {

    private int orderId;

    private String status;

    private List<String> products;

    private List<Integer> quantities;

    public Order(
            int orderId,
            String status,
            List<String> products,
            List<Integer> quantities) {

        this.orderId = orderId;

        this.status = status;

        this.products = products;

        this.quantities = quantities;
    }

    public void setStatus(String status) {

        this.status = status;
    }

    public void getDetails() {

        System.out.println(
                "Order ID: " + orderId);

        System.out.println(
                "Status: " + status);
    }
}
